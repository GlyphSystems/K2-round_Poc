// ---------- PHASE 6B: STATE DESYNC VALIDATION ----------

println!("--- PHASE 6B: STATE DESYNC VALIDATION ---");

let cycles: u32 = 10;

let borrow_pattern: [u128; 3] = [
    1_000_000_001,
    1_000_000_003,
    1_000_000_007,
];

let repay_pattern: [u128; 3] = [
    1_000_000_000,
    1_000_000_002,
    1_000_000_006,
];

println!("cycles: {:?}", cycles);

let mut boundary_hit = false;

for i in 0..cycles {
    let index = (i % 3) as usize;
    let borrow_amount = borrow_pattern[index];
    let repay_amount = repay_pattern[index];

    println!("================================================");
    println!("--- PHASE 6B CYCLE {:?} ---", i + 1);

    setup.router.borrow(
        &setup.user,
        &setup.asset_b,
        &borrow_amount,
        &1u32,
        &0u32,
        &setup.user,
    );

    println!("borrow succeeded");

    let repay_result = setup.router.try_repay(
        &setup.user,
        &setup.asset_b,
        &repay_amount,
        &1u32,
        &setup.user,
    );

    match repay_result {
        Ok(_) => {
            println!("repay succeeded");
        }
        Err(err) => {
            boundary_hit = true;

            println!("--- FAILURE BOUNDARY DETECTED ---");
            println!("cycle: {:?}", i + 1);
            println!("repay failure: {:?}", err);

            break;
        }
    }
}

assert!(
    boundary_hit,
    "expected repayment boundary failure was not reached"
);

println!("================================================");
println!("--- STATE DESYNC VALIDATION OPERATIONS ---");

let before_validation = setup.router.get_user_account_data(&setup.user);

println!("before collateral: {:?}", before_validation.total_collateral_base);
println!("before debt: {:?}", before_validation.total_debt_base);
println!("before health factor: {:?}", before_validation.health_factor);

let small_repay: u128 = 1;

println!("--- SMALL REPAY TEST ---");

let small_repay_result = setup.router.try_repay(
    &setup.user,
    &setup.asset_b,
    &small_repay,
    &1u32,
    &setup.user,
);

println!("small_repay_result: {:?}", small_repay_result);

let micro_borrow: u128 = 1;

println!("--- MICRO BORROW TEST ---");

let micro_borrow_result = setup.router.try_borrow(
    &setup.user,
    &setup.asset_b,
    &micro_borrow,
    &1u32,
    &0u32,
    &setup.user,
);

println!("micro_borrow_result: {:?}", micro_borrow_result);

let after_validation = setup.router.get_user_account_data(&setup.user);

println!("================================================");
println!("--- FINAL VALIDATION TELEMETRY ---");

println!("final collateral: {:?}", after_validation.total_collateral_base);
println!("final debt: {:?}", after_validation.total_debt_base);
println!("final health factor: {:?}", after_validation.health_factor);

println!(
    "debt delta: {:?}",
    after_validation
        .total_debt_base
        .saturating_sub(before_validation.total_debt_base)
);

println!(
    "collateral delta: {:?}",
    after_validation
        .total_collateral_base
        .saturating_sub(before_validation.total_collateral_base)
);
