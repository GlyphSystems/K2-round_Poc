// ---------- PHASE 6D: CROSS-OPERATION PERSISTENCE VALIDATION ----------

println!("--- PHASE 6D: CROSS-OPERATION PERSISTENCE VALIDATION ---");

let cycles: u32 = 10;

let borrow_pattern: [u128; 3] = [
    1_000_000_001,
    1_000_000_003,
    1_000_000_007,
];

let repay_pattern: [u128; 3] = [
    1_000_000_000,
    1_000_000_002,
    1_000_000_006,
];

let mut boundary_hit = false;

for i in 0..cycles {
    let index = (i % 3) as usize;

    let borrow_amount = borrow_pattern[index];
    let repay_amount = repay_pattern[index];

    println!("================================================");
    println!("--- PHASE 6D CYCLE {:?} ---", i + 1);

    setup.router.borrow(
        &setup.user,
        &setup.asset_b,
        &borrow_amount,
        &1u32,
        &0u32,
        &setup.user,
    );

    println!("borrow succeeded");

    let repay_result = setup.router.try_repay(
        &setup.user,
        &setup.asset_b,
        &repay_amount,
        &1u32,
        &setup.user,
    );

    match repay_result {
        Ok(_) => {
            println!("repay succeeded");
        }

        Err(err) => {
            boundary_hit = true;

            println!("--- FAILURE BOUNDARY DETECTED ---");
            println!("cycle: {:?}", i + 1);
            println!("repay failure: {:?}", err);

            break;
        }
    }
}

assert!(
    boundary_hit,
    "expected repayment boundary failure was not reached"
);

println!("================================================");
println!("--- CROSS-OPERATION POST-FAILURE VALIDATION ---");

let before_ops =
    setup.router.get_user_account_data(&setup.user);

println!("before collateral: {:?}", before_ops.total_collateral_base);
println!("before debt: {:?}", before_ops.total_debt_base);
println!("before health factor: {:?}", before_ops.health_factor);
println!("before asset_a balance: {:?}", setup.asset_a_token.balance(&setup.user));
println!("before asset_b balance: {:?}", setup.asset_b_token.balance(&setup.user));
